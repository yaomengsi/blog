    fast-syntax-highlighting
    direnv

unset_proxy() {
    unset http_proxy
    unset HTTP_PROXY
    unset https_proxy
    unset HTTPS_PROXY
    unset all_proxy
    unset ALL_PROXY
}

set_proxy() {
    echo "host ip:" "${HOST_IP}"
    echo "wsl ip:" "${WSL_IP}"
    proxy_http="${HOST_PROXY_HTTP}"
    export http_proxy="${proxy_http}"
    export HTTP_PROXY="${proxy_http}"
    export https_proxy="${proxy_http}"
    export HTTPS_PROXY="${proxy_http}"
    export all_proxy="${proxy_http}"
    export ALL_PROXY="${proxy_http}"
    echo "proxy:" "${proxy_http}"
}
